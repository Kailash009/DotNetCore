>> In between "Key-value" , It must be associated with : (Colon) Symbol.

For Example:-
----------------------------
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JSON Data Types</title>
</head>
<body>
    <script>
        let myData=
        {
            "name":"Satish",
            "age":23,
            "isActive":true,
            "Salary":74567.567,   // Number
            "Interest":["Java","Dot Net","Php","Python"]
        }
        console.log(myData);
        console.log(myData.name);
        console.log(typeof(myData.name));
        console.log(myData.age);
        console.log(typeof(myData.age));
        console.log(myData.isActive);
        console.log(typeof(myData.isActive));
        console.log(myData.Salary);
        console.log(typeof(myData.Salary));
        console.log(myData.Interest);
        console.log(typeof(myData.Interest));
        console.log(myData.Interest[2]); // Php // To access particular value of array.
        console.log(myData.Interest[0]); // JAVA
        for(x in myData.Interest) // x=0,x=1,x=2,x=3 // To access all values of array. 
        {
            console.log(x+"="+myData.Interest[x]);   // myData.Interest[0] // Java,Dot Net,Php,Python
        }
    </script>
</body>
</html>
